98  {
99  switch (true) {
100  case is_resource($value):
101  if ($exceptionOnInvalidType) {
102  throw new DumpException(sprintf('Unable to dump PHP resources in a YAML file ("%s").', get_resource_type($value)));
103  }
104 
105  return 'null';
106  case is_object($value):
107  if ($objectSupport) {
108  return '!!php/object:'.serialize($value);
109  }
110 
111  if ($exceptionOnInvalidType) {
112  throw new DumpException('Object support when dumping a YAML file has been disabled.');
113  }
114 
115  return 'null';
116  case is_array($value):
117  return self::dumpArray($value, $exceptionOnInvalidType, $objectSupport);
118  case null === $value:
119  return 'null';
120  case true === $value:
121  return 'true';
122  case false === $value:
123  return 'false';
124  case ctype_digit($value):
125  return is_string($value) ? "'$value'" : (int) $value;
126  case is_numeric($value):
127  $locale = setlocale(LC_NUMERIC, 0);
128  if (false !== $locale) {
129  setlocale(LC_NUMERIC, 'C');
130  }
131  if (is_float($value)) {
132  $repr = (string) $value;
133  if (is_infinite($value)) {
134  $repr = str_ireplace('INF', '.Inf', $repr);
135  } elseif (floor($value) == $value && $repr == $value) {
136  // Preserve float data type since storing a whole number will result in integer value.
137  $repr = '!!float '.$repr;
138  }
139  } else {
140  $repr = is_string($value) ? "'$value'" : (string) $value;
141  }
142  if (false !== $locale) {
143  setlocale(LC_NUMERIC, $locale);
144  }
145 
146  return $repr;
147  case '' == $value:
148  return "''";
149  case Escaper::requiresDoubleQuoting($value):
150  return Escaper::escapeWithDoubleQuotes($value);
151  case Escaper::requiresSingleQuoting($value):
152  case preg_match(self::getHexRegex(), $value):
153  case preg_match(self::getTimestampRegex(), $value):

170  {
171  // array
172  $keys = array_keys($value);
173  $keysCount = count($keys);
174  if ((1 === $keysCount && '0' == $keys[0])
175  || ($keysCount > 1 && array_reduce($keys, function ($v, $w) { return (int) $v + $w; }, 0) === $keysCount * ($keysCount - 1) / 2)
176  ) {
177  $output = array();
178  foreach ($value as $val) {
179  $output[] = self::dump($val, $exceptionOnInvalidType, $objectSupport);
180  }

183  }
184 
185  // mapping

210  public static function parseScalar($scalar, $delimiters = null, $stringDelimiters = array('"', "'"), &$i = 0, $evaluate = true, $references = array())
211  {
212  if (in_array($scalar[$i], $stringDelimiters)) {
213  // quoted scalar
214  $output = self::parseQuotedScalar($scalar, $i);
215 
216  if (null !== $delimiters) {
217  $tmp = ltrim(substr($scalar, $i), ' ');
218  if (!in_array($tmp[0], $delimiters)) {
219  throw new ParseException(sprintf('Unexpected characters (%s).', substr($scalar, $i)));
220  }
221  }
222  } else {
223  // "normal" string
224  if (!$delimiters) {
225  $output = substr($scalar, $i);
226  $i += strlen($output);
227 
228  // remove comments
229  if (preg_match('/[ \t]+#/', $output, $match, PREG_OFFSET_CAPTURE)) {
230  $output = substr($output, 0, $match[0][1]);
231  }
232  } elseif (preg_match('/^(.+?)('.implode('|', $delimiters).')/', substr($scalar, $i), $match)) {
233  $output = $match[1];
234  $i += strlen($output);
235  } else {
236  throw new ParseException(sprintf('Malformed inline YAML string (%s).', $scalar));
237  }
238 
239  // a non-quoted string cannot start with @ or ` (reserved) nor with a scalar indicator (| or >)
240  if ($output && ('@' === $output[0] || '`' === $output[0] || '|' === $output[0] || '>' === $output[0])) {
241  @trigger_error(sprintf('Not quoting the scalar "%s" starting with "%s" is deprecated since Symfony 2.8 and will throw a ParseException in 3.0.', $output, $output[0]), E_USER_DEPRECATED);
242 
243  // to be thrown in 3.0
244  // throw new ParseException(sprintf('The reserved indicator "%s" cannot start a plain scalar; you need to quote the scalar.', $output[0]));
245  }
246 
247  if ($evaluate) {
248  $output = self::evaluateScalar($output, $references);
249  }
250  }
251 
252  return $output;
253  }

296  private static function parseSequence($sequence, &$i = 0, $references = array())
297  {
298  $output = array();
299  $len = strlen($sequence);
300  ++$i;
301 
302  // [foo, bar, ...]
303  while ($i < $len) {
304  switch ($sequence[$i]) {
305  case '[':
306  // nested sequence
307  $output[] = self::parseSequence($sequence, $i, $references);
308  break;
309  case '{':
310  // nested mapping
311  $output[] = self::parseMapping($sequence, $i, $references);
312  break;
313  case ']':
314  return $output;
315  case ',':
316  case ' ':
317  break;
318  default:
319  $isQuoted = in_array($sequence[$i], array('"', "'"));
320  $value = self::parseScalar($sequence, array(',', ']'), array('"', "'"), $i, true, $references);
321 
322  // the value can be an array if a reference has been resolved to an array var
323  if (!is_array($value) && !$isQuoted && false !== strpos($value, ': ')) {
324  // embedded mapping?
325  try {
326  $pos = 0;
327  $value = self::parseMapping('{'.$value.'}', $pos, $references);
328  } catch (\InvalidArgumentException $e) {
329  // no, it's not
330  }
331  }
332 
333  $output[] = $value;
334 
335  --$i;
336  }
337 
338  ++$i;
339  }
340 
341  throw new ParseException(sprintf('Malformed inline YAML string %s', $sequence));

356  {
357  $output = array();

359  ++$i;
360 
361  // {foo: bar, bar:foo, ...}
362  while ($i < $len) {
363  switch ($mapping[$i]) {
364  case ' ':
365  case ',':
366  ++$i;
367  continue 2;
368  case '}':
369  if (self::$objectForMap) {
370  return (object) $output;
371  }
372 
373  return $output;
374  }
375 
376  // key
377  $key = self::parseScalar($mapping, array(':', ' '), array('"', "'"), $i, false);
378 
379  // value
380  $done = false;
381 
382  while ($i < $len) {
383  switch ($mapping[$i]) {
384  case '[':
385  // nested sequence
386  $value = self::parseSequence($mapping, $i, $references);
387  // Spec: Keys MUST be unique; first one wins.
388  // Parser cannot abort this mapping earlier, since lines
389  // are processed sequentially.
390  if (!isset($output[$key])) {
391  $output[$key] = $value;
392  }
393  $done = true;
394  break;
395  case '{':
396  // nested mapping
397  $value = self::parseMapping($mapping, $i, $references);
398  // Spec: Keys MUST be unique; first one wins.
399  // Parser cannot abort this mapping earlier, since lines
400  // are processed sequentially.
401  if (!isset($output[$key])) {
402  $output[$key] = $value;
403  }
404  $done = true;
405  break;
406  case ':':
407  case ' ':
408  break;
409  default:
410  $value = self::parseScalar($mapping, array(',', '}'), array('"', "'"), $i, true, $references);

412  // Parser cannot abort this mapping earlier, since lines
413  // are processed sequentially.
414  if (!isset($output[$key])) {
415  $output[$key] = $value;
416  }
417  $done = true;

419  }

442  private static function evaluateScalar($scalar, $references = array())
443  {
444  $scalar = trim($scalar);
445  $scalarLower = strtolower($scalar);
446 
447  if (0 === strpos($scalar, '*')) {
448  if (false !== $pos = strpos($scalar, '#')) {
449  $value = substr($scalar, 1, $pos - 2);
450  } else {
451  $value = substr($scalar, 1);
452  }
453 
454  // an unquoted *
455  if (false === $value || '' === $value) {
456  throw new ParseException('A reference must contain at least one character.');
457  }
458 
459  if (!array_key_exists($value, $references)) {
460  throw new ParseException(sprintf('Reference "%s" does not exist.', $value));
461  }
462 
463  return $references[$value];
464  }
465 
466  switch (true) {
467  case 'null' === $scalarLower:
468  case '' === $scalar:
469  case '~' === $scalar:
470  return;
471  case 'true' === $scalarLower:
472  return true;
473  case 'false' === $scalarLower:
474  return false;
475  // Optimise for returning strings.
476  case $scalar[0] === '+' || $scalar[0] === '-' || $scalar[0] === '.' || $scalar[0] === '!' || is_numeric($scalar[0]):
477  switch (true) {
478  case 0 === strpos($scalar, '!str'):
479  return (string) substr($scalar, 5);
480  case 0 === strpos($scalar, '! '):
481  return (int) self::parseScalar(substr($scalar, 2));
482  case 0 === strpos($scalar, '!!php/object:'):
483  if (self::$objectSupport) {
484  return unserialize(substr($scalar, 13));
485  }
486 
487  if (self::$exceptionOnInvalidType) {
488  throw new ParseException('Object support when parsing a YAML file has been disabled.');
489  }
490 
491  return;
492  case 0 === strpos($scalar, '!!float '):
493  return (float) substr($scalar, 8);
494  case ctype_digit($scalar):
495  $raw = $scalar;
496  $cast = (int) $scalar;
497 
498  return '0' == $scalar[0] ? octdec($scalar) : (((string) $raw == (string) $cast) ? $cast : $raw);
499  case '-' === $scalar[0] && ctype_digit(substr($scalar, 1)):
500  $raw = $scalar;
501  $cast = (int) $scalar;
502 
503  return '0' == $scalar[1] ? octdec($scalar) : (((string) $raw === (string) $cast) ? $cast : $raw);
504  case is_numeric($scalar):
505  case preg_match(self::getHexRegex(), $scalar):
506  return '0x' === $scalar[0].$scalar[1] ? hexdec($scalar) : (float) $scalar;
507  case '.inf' === $scalarLower:
508  case '.nan' === $scalarLower:
509  return -log(0);
510  case '-.inf' === $scalarLower:
511  return log(0);
512  case preg_match('/^(-|\+)?[0-9,]+(\.[0-9]+)?$/', $scalar):
513  return (float) str_replace(',', '', $scalar);
514  case preg_match(self::getTimestampRegex(), $scalar):
515  return strtotime($scalar);
516  }
517  default:
518  return (string) $scalar;
519  }
520  }
